import React, { useState, useEffect, useMemo } from "react"; import { View, Text, TextInput, Button, ScrollView } from "react-native"; import AsyncStorage from "@react-native-async-storage/async-storage"; export default function App() { const getToday = () => new Date().toISOString().split("T")[0]; const [input, setInput] = useState(""); const [weightInput, setWeightInput] = useState(""); const [data, setData] = useState({ foodLogs: {}, weightLogs: {}, waterLogs: {}, savedMeals: [], goals: { calories: 1400, protein: 125, carbs: 140, fat: 45, water: 80 }, streak: 0, lastLoggedDate: null }); // LOAD DATA useEffect(() => { const load = async () => { const saved = await AsyncStorage.getItem("fitnessData"); if (saved) setData(JSON.parse(saved)); }; load(); }, []); // SAVE DATA useEffect(() => { AsyncStorage.setItem("fitnessData", JSON.stringify(data)); }, [data]); // ADD FOOD const addFood = (food) => { const today = getToday(); setData(prev => { const isNewDay = prev.lastLoggedDate !== today; return { ...prev, foodLogs: { ...prev.foodLogs, [today]: [...(prev.foodLogs[today] || []), food] }, streak: isNewDay ? prev.streak + 1 : prev.streak, lastLoggedDate: today }; }); }; // QUICK ADD PARSER const quickAdd = () => { const text = input.toLowerCase(); if (text.includes("chicken")) { addFood({ name: "Chicken", calories: 200, protein: 30, carbs: 0, fat: 5 }); } if (text.includes("rice")) { addFood({ name: "Rice", calories: 180, protein: 4, carbs: 40, fat: 1 }); } if (text.includes("egg")) { addFood({ name: "Eggs", calories: 140, protein: 12, carbs: 1, fat: 10 }); } setInput(""); }; // LOG WEIGHT const logWeight = () => { const today = getToday(); setData(prev => ({ ...prev, weightLogs: { ...prev.weightLogs, [today]: parseFloat(weightInput) } })); setWeightInput(""); }; // WATER const logWater = (oz) => { const today = getToday(); setData(prev => ({ ...prev, waterLogs: { ...prev.waterLogs, [today]: (prev.waterLogs[today] || 0) + oz } })); }; // MACROS const getMacros = (foods = []) => { return foods.reduce((t, f) => ({ calories: t.calories + f.calories, protein: t.protein + f.protein, carbs: t.carbs + f.carbs, fat: t.fat + f.fat }), { calories: 0, protein: 0, carbs: 0, fat: 0 }); }; const today = getToday(); const todaysFood = useMemo(() => { return data.foodLogs[today] || []; }, [data.foodLogs]); const totals = getMacros(todaysFood); const remaining = { calories: data.goals.calories - totals.calories, protein: data.goals.protein - totals.protein, carbs: data.goals.carbs - totals.carbs, fat: data.goals.fat - totals.fat }; // SMART ACTION const getDailyAction = () => { if (!todaysFood.length) return "Log your first meal"; if (remaining.protein > 30) return "Eat protein 🍗"; if (remaining.calories < 200) return "Almost done — go light"; if ((data.waterLogs[today] || 0) < 40) return "Drink water 💧"; return "You're on track 🔥"; }; // MEAL SUGGESTION const generateMeal = () => { if (remaining.protein > 40) return "Chicken + yogurt"; if (remaining.carbs > 40) return "Rice + fruit"; return "Light snack"; }; // WEIGHT PROJECTION const getWeightTrend = () => { const entries = Object.values(data.weightLogs); if (entries.length < 2) return null; const start = entries[0]; const end = entries[entries.length - 1]; const rate = (start - end) / entries.length; if (rate <= 0) return null; return Math.round((end - 120) / rate); }; return ( 🔥 GOD MODE TRACKER {/* DAILY ACTION */} {getDailyAction()} {/* INPUT */}
